11/// In traditional color spaces, a color can be identified solely by the
12/// observer's measurement of the color.
13///
14/// Color appearance models such as CAM16 also use information about the
15/// environment where the color was observed, known as the viewing conditions.
16///
17/// For example, white under the traditional assumption of a midday sun white
18/// point is accurately measured as a slightly chromatic blue by CAM16.
19/// (roughly, hue 203, chroma 3, lightness 100)
20///
21/// This struct caches intermediate values of the CAM16 conversion process that
22/// depend only on viewing conditions, enabling speed ups.
23#[derive(Debug)]
24pub struct ViewingConditions {
25    pub white_point: [f64; 3],
26    pub adapting_luminance: f64,
27    pub background_lstar: f64,
28    pub surround: f64,
29    pub discounting_illuminant: bool,
30
31    pub background_ytowhite_point_y: f64,
32    pub aw: f64,
33    pub nbb: f64,
34    pub ncb: f64,
35    pub c: f64,
36    pub n_c: f64,
37    pub drgb_inverse: [f64; 3],
38    pub rgb_d: [f64; 3],
39    pub fl: f64,
40    pub f_lroot: f64,
41    pub z: f64,
42}
43
44impl ViewingConditions {
45    pub fn s_rgb() -> Self {
46        Self::make(None, None, None, None, None)
47    }
48
49    /// Convenience constructor for [`ViewingConditions`].
50    ///
51    /// # Parameters affecting color appearance include:
52    /// * `whitePoint` - coordinates of white in Xyz color space.
53    /// * `adaptingLuminance` - light strength, in lux.
54    /// * `backgroundLstar` - average luminance of 10 degrees around color.
55    /// * `surround` - brightness of the entire environment.
56    /// * `discountingIlluminant` - whether eyes have adjusted to lighting.
57    ///
58    /// # Panics
59    ///
60    /// Will panic if the surround is equal or between 0.0 and 2.0
61    pub fn make(
62        white_point: Option<[f64; 3]>,
63        adapting_luminance: Option<f64>,
64        background_lstar: Option<f64>,
65        surround: Option<f64>,
66        discounting_illuminant: Option<bool>,
67    ) -> Self {
68        let white_point = white_point.unwrap_or(WHITE_POINT_D65);
69        let adapting_luminance = adapting_luminance.unwrap_or(-1.0);
70        let background_lstar = background_lstar.unwrap_or(50.0);
71        let surround = surround.unwrap_or(2.0);
72        let discounting_illuminant = discounting_illuminant.unwrap_or(false);
73
74        let adapting_luminance = if adapting_luminance > 0.0 {
75            adapting_luminance
76        } else {
77            200.0 / PI * y_from_lstar(50.0) / 100.0
78        };
79        // A background of pure black is non-physical and leads to infinities that
80        // represent the idea that any color viewed in pure black can't be seen.
81        let background_lstar = (0.1_f64).max(background_lstar);
82        // Transform test illuminant white in Xyz to 'cone'/'rgb' responses
83        let xyz = white_point;
84        let (r_w, g_w, b_w) = (
85            xyz[2].mul_add(-0.051461, xyz[0].mul_add(0.401288, xyz[1] * 0.650173)),
86            xyz[2].mul_add(0.045854, xyz[0].mul_add(-0.250268, xyz[1] * 1.204414)),
87            xyz[2].mul_add(0.953127, xyz[0].mul_add(-0.002079, xyz[1] * 0.048952)),
88        );
89
90        // Scale input surround, domain (0, 2), to CAM16 surround, domain (0.8, 1.0)
91        assert!((0.0..=2.0).contains(&surround));
92
93        let f = 0.8 + (surround / 10.0);
94        // "Exponential non-linearity"
95        let c = if f >= 0.9 {
96            lerp(0.59, 0.69, (f - 0.9) * 10.0)
97        } else {
98            lerp(0.525, 0.59, (f - 0.8) * 10.0)
99        };
100        // Calculate degree of adaptation to illuminant
101        let d = if discounting_illuminant {
102            1.0
103        } else {
104            f * (1.0f64 / 3.6f64).mul_add(-((-adapting_luminance - 42.0) / 92.0).exp(), 1.0)
105        };
106        // Per Li et al, if D is greater than 1 or less than 0, set it to 1 or 0.
107        let d = d.clamp(0.0, 1.0);
108        // chromatic induction factor
109        let nc = f;
110
111        // Cone responses to the whitePoint, r/g/b/W, adjusted for discounting.
112        //
113        // Why use 100.0 instead of the white point's relative luminance?
114        //
115        // Some papers and implementations, for both CAM02 and CAM16, use the Y
116        // value of the reference white instead of 100. Fairchild's Color Appearance
117        // Models (3rd edition) notes that this is in error: it was included in the
118        // CIE 2004a report on CIECAM02, but, later parts of the conversion process
119        // account for scaling of appearance relative to the white point relative
120        // luminance. This part should simply use 100 as luminance.
121        let rgb_d = [
122            d.mul_add(100.0 / r_w, 1.0) - d,
123            d.mul_add(100.0 / g_w, 1.0) - d,
124            d.mul_add(100.0 / b_w, 1.0) - d,
125        ];
126
127        // Factor used in calculating meaningful factors
128        let k = 1.0 / 5.0f64.mul_add(adapting_luminance, 1.0);
129        let k4 = k * k * k * k; // pow(k, 4)
130        let k4_f = 1.0 - k4;
131
132        // Luminance-level adaptation factor
133        let fl = k4.mul_add(adapting_luminance, 0.1 * k4_f * k4_f * (5.0 * adapting_luminance).cbrt());
134        // Intermediate factor, ratio of background relative luminance to white relative
135        // luminance
136        let n = y_from_lstar(background_lstar) / white_point[1];
137
138        // Base exponential nonlinearity
139        // note Schlomer 2018 has a typo and uses 1.58, the correct factor is 1.48
140        let z = 1.48 + n.sqrt();
141
142        // Luminance-level induction factors
143        let nbb = 0.725 / n.powf(0.2);
144
145        let ncb = nbb;
146
147        // Discounted cone responses to the white point, adjusted for post-saturationtic
148        // adaptation perceptual nonlinearities.
149        let rgb_afactors = [
150            (fl * rgb_d[0] * r_w / 100.0).powf(0.42),
151            (fl * rgb_d[1] * g_w / 100.0).powf(0.42),
152            (fl * rgb_d[2] * b_w / 100.0).powf(0.42),
153        ];
154
155        let rgb_a = [
156            (400.0 * rgb_afactors[0]) / (rgb_afactors[0] + 27.13),
157            (400.0 * rgb_afactors[1]) / (rgb_afactors[1] + 27.13),
158            (400.0 * rgb_afactors[2]) / (rgb_afactors[2] + 27.13),
159        ];
160
161        let aw = (40.0f64.mul_add(rgb_a[0], 20.0 * rgb_a[1]) + rgb_a[2]) / 20.0 * nbb;
162
163        Self {
164            white_point,
165            adapting_luminance,
166            background_lstar,
167            surround,
168            discounting_illuminant,
169            background_ytowhite_point_y: n,
170            aw,
171            nbb,
172            ncb,
173            c,
174            n_c: nc,
175            drgb_inverse: [0.0, 0.0, 0.0],
176            rgb_d,
177            fl,
178            f_lroot: fl.powf(0.25),
179            z,
180        }
181    }
182}

192        let result1 = ViewingConditions::make(None, None, None, None, None);

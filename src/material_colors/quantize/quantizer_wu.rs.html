13// A histogram of all the input colors is constructed. It has the shape of a
14//  The cube would be too large if it contained all 16 million colors:
15// historical best practice is to use 5 bits  of the 8 in each channel,
16// reducing the histogram to a volume of ~32,000.
17const INDEX_BITS: u8 = 5;
18const BITS_TO_REMOVE: u8 = 8 - INDEX_BITS;
19const SIDE_LENGTH: usize = (1 << INDEX_BITS) + 1;
20const TOTAL_SIZE: usize = SIDE_LENGTH.pow(3);
21
22pub struct QuantizerWu {
23    weights: Vec<i64>,
24    moments_r: Vec<i64>,
25    moments_g: Vec<i64>,
26    moments_b: Vec<i64>,
27    moments: Vec<f64>,
28    cubes: Vec<Cube>,
29}

50impl Quantizer for QuantizerWu {
51    fn quantize(pixels: &[Rgb], max_colors: usize) -> QuantizerResult {
52        let mut result = QuantizerMap::quantize(pixels, max_colors);
53
54        result.color_to_count.sort_by(|_, a, _, b| a.cmp(b));
55
56        let mut quantizer = Self::new(max_colors);
57
58        quantizer.construct_histogram(result.color_to_count);
59        quantizer.compute_moments();
60
61        let create_boxes_result = quantizer.create_boxes(max_colors);
62        let color_to_count = quantizer.create_result(create_boxes_result.result_count);
63
64        QuantizerResult {
65            color_to_count,
66            input_pixel_to_cluster_pixel: IndexMap::default(),
67        }
68    }
69}

71impl QuantizerWu {
72    pub fn get_index<T: Into<usize>>(r: T, g: T, b: T) -> usize {
73        let r: usize = r.into();
74        let g: usize = g.into();
75        let b: usize = b.into();
76
77        (r << (INDEX_BITS * 2)) + (r << (INDEX_BITS + 1)) + (g << INDEX_BITS) + r + g + b
78    }
79
80    pub fn construct_histogram(&mut self, pixels: IndexMap<Rgb, u32>) {
81        for (pixel, count) in pixels {
82            let red = pixel.red;
83            let green = pixel.green;
84            let blue = pixel.blue;
85
86            let i_r = (red >> BITS_TO_REMOVE) + 1;
87            let i_g = (green >> BITS_TO_REMOVE) + 1;
88            let i_b = (blue >> BITS_TO_REMOVE) + 1;
89
90            let index = Self::get_index(i_r, i_g, i_b);
91
92            self.weights[index] += i64::from(count);
93
94            self.moments_r[index] += i64::from(red) * i64::from(count);
95            self.moments_g[index] += i64::from(green) * i64::from(count);
96            self.moments_b[index] += i64::from(blue) * i64::from(count);
97
98            self.moments[index] +=
99                f64::from(count) * f64::from(blue).mul_add(f64::from(blue), f64::from(red).mul_add(f64::from(red), f64::from(green) * f64::from(green)));
100        }
101    }

103    pub fn compute_moments(&mut self) {
104        for r in 1..SIDE_LENGTH {
105            let mut area = [0; SIDE_LENGTH];
106            let mut area_r = [0; SIDE_LENGTH];
107            let mut area_g = [0; SIDE_LENGTH];
108            let mut area_b = [0; SIDE_LENGTH];
109            let mut area2 = [0.0; SIDE_LENGTH];
110
111            for g in 1..SIDE_LENGTH {
112                let mut line = 0;
113                let mut line_r = 0;
114                let mut line_g = 0;
115                let mut line_b = 0;
116                let mut line2 = 0.0;
117
118                for b in 1..SIDE_LENGTH {
119                    let index = Self::get_index(r, g, b);
120
121                    line += self.weights[index];
122                    line_r += self.moments_r[index];
123                    line_g += self.moments_g[index];
124                    line_b += self.moments_b[index];
125                    line2 += self.moments[index];
126
127                    area[b] += line;
128                    area_r[b] += line_r;
129                    area_g[b] += line_g;
130                    area_b[b] += line_b;
131                    area2[b] += line2;
132
133                    let previous_index = Self::get_index(r - 1, g, b);
134
135                    self.weights[index] = self.weights[previous_index] + area[b];
136                    self.moments_r[index] = self.moments_r[previous_index] + area_r[b];
137                    self.moments_g[index] = self.moments_g[previous_index] + area_g[b];
138                    self.moments_b[index] = self.moments_b[previous_index] + area_b[b];
139                    self.moments[index] = self.moments[previous_index] + area2[b];
140                }
141            }
142        }
143    }

145    pub fn create_boxes(&mut self, max_color_count: usize) -> CreateBoxesResult {
146        self.cubes[0] = Cube {
147            pixels: [Rgb::default(), Rgb::new(SIDE_LENGTH as u8 - 1, SIDE_LENGTH as u8 - 1, SIDE_LENGTH as u8 - 1)],
148            vol: 0,
149        };
150
151        let mut volume_variance = vec![0.0; max_color_count];
152        let mut next = 0;
153        let mut generated_color_count = max_color_count;
154        let mut i = 1;
155
156        while i < max_color_count {
157            if self.cut(next, i) {
158                volume_variance[next] = if self.cubes[next].vol > 1 { self.variance(&self.cubes[next]) } else { 0.0 };
159
160                volume_variance[i] = if self.cubes[i].vol > 1 { self.variance(&self.cubes[i]) } else { 0.0 };
161            } else {
162                volume_variance[next] = 0.0;
163
164                i -= 1;
165            }
166
167            next = 0;
168
169            let mut temp = volume_variance[0];
170
171            let mut j = 1;
172
173            while j <= i {
174                if volume_variance[j] > temp {
175                    temp = volume_variance[j];
176                    next = j;
177                }
178
179                j += 1;
180            }
181
182            if temp <= 0.0 {
183                generated_color_count = i + 1;
184
185                break;
186            }
187
188            i += 1;
189        }
190
191        CreateBoxesResult {
192            requested_count: max_color_count,
193            result_count: generated_color_count,
194        }
195    }

197    pub fn create_result(&self, color_count: usize) -> IndexMap<Rgb, u32> {
198        let mut result = IndexMap::default();
199
200        for i in 0..color_count {
201            let cube = &self.cubes[i];
202            let weight = Self::volume(cube, &self.weights);
203
204            if weight > 0 {
205                let r = ((Self::volume(cube, &self.moments_r)) / weight) as u8;
206                let g = ((Self::volume(cube, &self.moments_g)) / weight) as u8;
207                let b = ((Self::volume(cube, &self.moments_b)) / weight) as u8;
208
209                result.insert(Rgb::new(r, g, b), 0);
210            }
211        }
212
213        result
214    }

216    pub fn variance(&self, cube: &Cube) -> f64 {
217        let dr = Self::volume(cube, &self.moments_r) as f64;
218        let dg = Self::volume(cube, &self.moments_g) as f64;
219        let db = Self::volume(cube, &self.moments_b) as f64;
220
221        let xx = self.moments[Self::get_index::<u8>(cube.r(1), cube.g(1), cube.b(1))]
222            - self.moments[Self::get_index::<u8>(cube.r(1), cube.g(1), cube.b(0))]
223            - self.moments[Self::get_index::<u8>(cube.r(1), cube.g(0), cube.b(1))]
224            + self.moments[Self::get_index::<u8>(cube.r(1), cube.g(0), cube.b(0))]
225            - self.moments[Self::get_index::<u8>(cube.r(0), cube.g(1), cube.b(1))]
226            + self.moments[Self::get_index::<u8>(cube.r(0), cube.g(1), cube.b(0))]
227            + self.moments[Self::get_index::<u8>(cube.r(0), cube.g(0), cube.b(1))]
228            - self.moments[Self::get_index::<u8>(cube.r(0), cube.g(0), cube.b(0))];
229
230        let hypotenuse = db.mul_add(db, dr.mul_add(dr, dg * dg));
231        let volume = Self::volume(cube, &self.weights) as f64;
232
233        xx - (hypotenuse / volume)
234    }

236    pub fn cut(&mut self, next: usize, i: usize) -> bool {
237        let (mut one, mut two) = (self.cubes[next].clone(), self.cubes[i].clone());
238
239        let whole_r = Self::volume(&one, &self.moments_r);
240        let whole_g = Self::volume(&one, &self.moments_g);
241        let whole_b = Self::volume(&one, &self.moments_b);
242        let whole_w = Self::volume(&one, &self.weights);
243
244        let max_rresult = self.maximize(&one, &Direction::Red, one.r::<i32>(0) + 1, one.r::<i32>(1), whole_r, whole_g, whole_b, whole_w);
245        let max_gresult = self.maximize(
246            &one,
247            &Direction::Green,
248            one.g::<i32>(0) + 1,
249            one.g::<i32>(1),
250            whole_r,
251            whole_g,
252            whole_b,
253            whole_w,
254        );
255        let max_bresult = self.maximize(&one, &Direction::Blue, one.b::<i32>(0) + 1, one.b::<i32>(1), whole_r, whole_g, whole_b, whole_w);
256
257        let cut_direction: Direction;
258
259        let max_r = max_rresult.maximum;
260        let max_g = max_gresult.maximum;
261        let max_b = max_bresult.maximum;
262
263        if max_r >= max_g && max_r >= max_b {
264            cut_direction = Direction::Red;
265
266            if max_rresult.cut_location < 0 {
267                return false;
268            }
269        } else if max_g >= max_r && max_g >= max_b {
270            cut_direction = Direction::Green;
271        } else {
272            cut_direction = Direction::Blue;
273        }
274
275        two.pixels[1].red = one.pixels[1].red;
276        two.pixels[1].green = one.pixels[1].green;
277        two.pixels[1].blue = one.pixels[1].blue;
278
279        match cut_direction {
280            Direction::Red => {
281                one.pixels[1].red = max_rresult.cut_location as u8;
282                two.pixels[0].red = one.pixels[1].red;
283                two.pixels[0].green = one.pixels[0].green;
284                two.pixels[0].blue = one.pixels[0].blue;
285            }
286            Direction::Green => {
287                one.pixels[1].green = max_gresult.cut_location as u8;
288                two.pixels[0].red = one.pixels[0].red;
289                two.pixels[0].green = one.pixels[1].green;
290                two.pixels[0].blue = one.pixels[0].blue;
291            }
292            Direction::Blue => {
293                one.pixels[1].blue = max_bresult.cut_location as u8;
294                two.pixels[0].red = one.pixels[0].red;
295                two.pixels[0].green = one.pixels[0].green;
296                two.pixels[0].blue = one.pixels[1].blue;
297            }
298        }
299
300        one.vol = (one.r::<i32>(1) - one.r::<i32>(0)) * (one.g::<i32>(1) - one.g::<i32>(0)) * (one.b::<i32>(1) - one.b::<i32>(0));
301        two.vol = (two.r::<i32>(1) - two.r::<i32>(0)) * (two.g::<i32>(1) - two.g::<i32>(0)) * (two.b::<i32>(1) - two.b::<i32>(0));
302
303        self.cubes[next] = one;
304        self.cubes[i] = two;
305
306        true
307    }

309    pub fn maximize(
310        &self,
311        cube: &Cube,
312        direction: &Direction,
313        first: i32,
314        last: i32,
315        whole_r: i64,
316        whole_g: i64,
317        whole_b: i64,
318        whole_w: i64,
319    ) -> MaximizeResult {
320        let bottom_r = Self::bottom(cube, direction, &self.moments_r) as f64;
321        let bottom_g = Self::bottom(cube, direction, &self.moments_g) as f64;
322        let bottom_b = Self::bottom(cube, direction, &self.moments_b) as f64;
323        let bottom_w = Self::bottom(cube, direction, &self.weights) as f64;
324
325        let mut max = 0.0;
326        let mut cut = -1;
327
328        for i in first..last {
329            let mut half_r = bottom_r + Self::top(cube, direction, i, &self.moments_r) as f64;
330            let mut half_g = bottom_g + Self::top(cube, direction, i, &self.moments_g) as f64;
331            let mut half_b = bottom_b + Self::top(cube, direction, i, &self.moments_b) as f64;
332            let mut half_w = bottom_w + Self::top(cube, direction, i, &self.weights) as f64;
333
334            if half_w == 0.0 {
335                continue;
336            }
337
338            let mut temp_numerator = half_b.mul_add(half_b, half_r.mul_add(half_r, half_g.powi(2)));
339            let mut temp_denominator = half_w;
340            let mut temp = temp_numerator / temp_denominator;
341
342            half_r = whole_r as f64 - half_r;
343            half_g = whole_g as f64 - half_g;
344            half_b = whole_b as f64 - half_b;
345            half_w = whole_w as f64 - half_w;
346
347            if half_w == 0.0 {
348                continue;
349            }
350
351            temp_numerator = half_b.mul_add(half_b, half_r.mul_add(half_r, half_g.powi(2)));
352            temp_denominator = half_w;
353            temp += temp_numerator / temp_denominator;
354
355            if temp > max {
356                max = temp;
357                cut = i;
358            }
359        }
360
361        MaximizeResult {
362            cut_location: cut,
363            maximum: max,
364        }
365    }

367    pub fn volume(cube: &Cube, moment: &[i64]) -> i64 {
368        moment[Self::get_index::<u8>(cube.r(1), cube.g(1), cube.b(1))]
369            - moment[Self::get_index::<u8>(cube.r(1), cube.g(1), cube.b(0))]
370            - moment[Self::get_index::<u8>(cube.r(1), cube.g(0), cube.b(1))]
371            + moment[Self::get_index::<u8>(cube.r(1), cube.g(0), cube.b(0))]
372            - moment[Self::get_index::<u8>(cube.r(0), cube.g(1), cube.b(1))]
373            + moment[Self::get_index::<u8>(cube.r(0), cube.g(1), cube.b(0))]
374            + moment[Self::get_index::<u8>(cube.r(0), cube.g(0), cube.b(1))]
375            - moment[Self::get_index::<u8>(cube.r(0), cube.g(0), cube.b(0))]
376    }

378    pub fn bottom(cube: &Cube, direction: &Direction, moment: &[i64]) -> i64 {
379        match direction {
380            Direction::Red => {
381                -moment[Self::get_index::<u8>(cube.r(0), cube.g(1), cube.b(1))]
382                    + moment[Self::get_index::<u8>(cube.r(0), cube.g(1), cube.b(0))]
383                    + moment[Self::get_index::<u8>(cube.r(0), cube.g(0), cube.b(1))]
384                    - moment[Self::get_index::<u8>(cube.r(0), cube.g(0), cube.b(0))]
385            }
386            Direction::Green => {
387                -moment[Self::get_index::<u8>(cube.r(1), cube.g(0), cube.b(1))]
388                    + moment[Self::get_index::<u8>(cube.r(1), cube.g(0), cube.b(0))]
389                    + moment[Self::get_index::<u8>(cube.r(0), cube.g(0), cube.b(1))]
390                    - moment[Self::get_index::<u8>(cube.r(0), cube.g(0), cube.b(0))]
391            }
392            Direction::Blue => {
393                -moment[Self::get_index::<u8>(cube.r(1), cube.g(1), cube.b(0))]
394                    + moment[Self::get_index::<u8>(cube.r(1), cube.g(0), cube.b(0))]
395                    + moment[Self::get_index::<u8>(cube.r(0), cube.g(1), cube.b(0))]
396                    - moment[Self::get_index::<u8>(cube.r(0), cube.g(0), cube.b(0))]
397            }
398        }
399    }

401    pub fn top(cube: &Cube, direction: &Direction, position: i32, moment: &[i64]) -> i64 {
402        match direction {
403            Direction::Red => {
404                moment[Self::get_index(position as usize, cube.g(1), cube.b(1))]
405                    - moment[Self::get_index(position as usize, cube.g(1), cube.b(0))]
406                    - moment[Self::get_index(position as usize, cube.g(0), cube.b(1))]
407                    + moment[Self::get_index(position as usize, cube.g(0), cube.b(0))]
408            }
409            Direction::Green => {
410                moment[Self::get_index(cube.r(1), position as usize, cube.b(1))]
411                    - moment[Self::get_index(cube.r(1), position as usize, cube.b(0))]
412                    - moment[Self::get_index(cube.r(0), position as usize, cube.b(1))]
413                    + moment[Self::get_index(cube.r(0), position as usize, cube.b(0))]
414            }
415            Direction::Blue => {
416                moment[Self::get_index(cube.r(1), cube.g(1), position as usize)]
417                    - moment[Self::get_index(cube.r(1), cube.g(0), position as usize)]
418                    - moment[Self::get_index(cube.r(0), cube.g(1), position as usize)]
419                    + moment[Self::get_index(cube.r(0), cube.g(0), position as usize)]
420            }
421        }
422    }

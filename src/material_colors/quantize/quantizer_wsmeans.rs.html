18struct DistanceAndIndex {
19    distance: f64,
20    index: usize,
21}
22
23impl DistanceAndIndex {
24    pub const fn new(distance: f64, index: usize) -> Self {
25        Self { distance, index }
26    }
27}
28
29impl Eq for DistanceAndIndex {}
30impl PartialEq for DistanceAndIndex {
31    fn eq(&self, other: &Self) -> bool {
32        self.distance != other.distance
33    }
34}
35
36impl Ord for DistanceAndIndex {
37    fn cmp(&self, other: &Self) -> Ordering {
38        if self.distance < other.distance {
39            Ordering::Less
40        } else if self.distance > other.distance {
41            Ordering::Greater
42        } else {
43            Ordering::Equal
44        }
45    }
46}
47
48impl PartialOrd for DistanceAndIndex {
49    fn lt(&self, other: &Self) -> bool {
50        self.distance < other.distance
51    }
52
53    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
54        Some(self.cmp(other))
55    }
56}

72    pub fn quantize(input_pixels: &[Rgb], max_colors: usize, starting_clusters: &[Rgb]) -> QuantizerResult {
73        let mut pixel_to_count: IndexMap<Rgb, u32> = IndexMap::default();
74        let mut points: Vec<Lab> = vec![];
75        let mut pixels: Vec<Rgb> = vec![];
76
77        for input_pixel in input_pixels {
78            let pixel_count = pixel_to_count.get_mut(input_pixel);
79
80            if let Some(pixel_count) = pixel_count {
81                *pixel_count += 1;
82            } else {
83                pixels.push(*input_pixel);
84                points.push(PointProviderLab::lab_from_int(input_pixel));
85                pixel_to_count.insert(*input_pixel, 1);
86            }
87        }
88
89        let cluster_count = max_colors.min(points.len());
90
91        let mut clusters = starting_clusters.iter().map(PointProviderLab::lab_from_int).collect::<Vec<_>>();
92        let additional_clusters_needed = cluster_count - clusters.len();
93
94        if additional_clusters_needed > 0 {
95            let mut seed_generator = Random::new(0x42688);
96            let mut indices = vec![];
97
98            for _ in 0..additional_clusters_needed {
99                // Use existing points rather than generating random centroids.
100                //
101                // KMeans is extremely sensitive to initial clusters. This quantizer
102                // is meant to be used with a Wu quantizer that provides initial
103                // centroids, but Wu is very slow on unscaled images and when extracting
104                // more than 256 colors.
105                //
106                // Here, we can safely assume that more than 256 colors were requested
107                // for extraction. Generating random centroids tends to lead to many
108                // "empty" centroids, as the random centroids are nowhere near any pixels
109                // in the image, and the centroids from Wu are very refined and close
110                // to pixels in the image.
111                //
112                // Rather than generate random centroids, we'll pick centroids that
113                // are actual pixels in the image, and avoid duplicating centroids.
114
115                let mut index = seed_generator.next_range(points.len() as i32) as usize;
116
117                while indices.contains(&index) {
118                    index = seed_generator.next_range(points.len() as i32) as usize;
119                }
120
121                indices.push(index);
122            }
123
124            for index in indices {
125                clusters.push(points[index]);
126            }
127        }
128
129        #[cfg(feature = "std")]
130        Self::debug_log(format!("have {} starting clusters, {} points", clusters.len(), points.len()));
131
132        let mut cluster_indices = fill_array(points.len(), |index| index % cluster_count);
133        let mut index_matrix = vec![vec![0; cluster_count]; cluster_count];
134
135        let mut distance_to_index_matrix: Vec<Vec<DistanceAndIndex>> =
136            fill_array(cluster_count, |_| fill_array(cluster_count, |index| DistanceAndIndex::new(0.0, index)));
137        let mut pixel_count_sums = vec![0; cluster_count];
138
139        for iteration in 0..10 {
140            if Self::DEBUG {
141                for i in pixel_count_sums.iter_mut().take(cluster_count) {
142                    *i = 0;
143                }
144
145                for i in 0..points.len() {
146                    let cluster_index = cluster_indices[i];
147                    let count = pixel_to_count[&pixels[i]];
148
149                    pixel_count_sums[cluster_index] += count;
150                }
151
152                #[cfg(feature = "std")]
153                {
154                    let empty_clusters = pixel_count_sums
155                        .iter()
156                        .take(cluster_count)
157                        .filter(|pixel_count_sum| pixel_count_sum == &&0)
158                        .count();
159
160                    Self::debug_log(format!(
161                        "starting iteration {}; {empty_clusters} clusters are empty of {cluster_count}",
162                        iteration + 1
163                    ));
164                }
165            }
166
167            let mut points_moved = 0;
168
169            for i in 0..cluster_count {
170                for j in (i + 1)..cluster_count {
171                    let distance = PointProviderLab::distance(&clusters[i], &clusters[j]);
172
173                    distance_to_index_matrix[j][i].distance = distance;
174                    distance_to_index_matrix[j][i].index = i;
175                    distance_to_index_matrix[i][j].distance = distance;
176                    distance_to_index_matrix[i][j].index = j;
177                }
178
179                distance_to_index_matrix[i].sort();
180
181                for j in 0..cluster_count {
182                    index_matrix[i][j] = distance_to_index_matrix[i][j].index;
183                }
184            }
185
186            for i in 0..points.len() {
187                let point = points[i];
188                let previous_cluster_index = cluster_indices[i];
189                let previous_cluster = clusters[previous_cluster_index];
190                let previous_distance = PointProviderLab::distance(&point, &previous_cluster);
191
192                let mut minimum_distance = previous_distance;
193                let mut new_cluster_index = None;
194
195                for (j, cluster) in clusters.iter().enumerate().take(cluster_count) {
196                    if distance_to_index_matrix[previous_cluster_index][j].distance >= 4.0 * previous_distance {
197                        continue;
198                    }
199
200                    let distance = PointProviderLab::distance(&point, cluster);
201
202                    if distance < minimum_distance {
203                        minimum_distance = distance;
204                        new_cluster_index = Some(j);
205                    }
206                }
207
208                if let Some(new_cluster_index) = new_cluster_index {
209                    points_moved += 1;
210                    cluster_indices[i] = new_cluster_index;
211                }
212            }
213
214            if points_moved == 0 && iteration > 0 {
215                #[cfg(feature = "std")]
216                Self::debug_log(format!("terminated after {iteration} k-means iterations"));
217
218                break;
219            }
220
221            #[cfg(feature = "std")]
222            Self::debug_log(format!("iteration {} moved {points_moved}", iteration + 1));
223
224            let mut component_asums: Vec<f64> = vec![0.0; cluster_count];
225            let mut component_bsums: Vec<f64> = vec![0.0; cluster_count];
226            let mut component_csums: Vec<f64> = vec![0.0; cluster_count];
227
228            for pixel_count_sum in pixel_count_sums.iter_mut().take(cluster_count) {
229                *pixel_count_sum = 0;
230            }
231
232            for i in 0..points.len() {
233                let cluster_index = cluster_indices[i];
234                let point = points[i];
235                let count = pixel_to_count[&pixels[i]];
236
237                pixel_count_sums[cluster_index] += count;
238                component_asums[cluster_index] += point.l * f64::from(count);
239                component_bsums[cluster_index] += point.a * f64::from(count);
240                component_csums[cluster_index] += point.b * f64::from(count);
241            }
242
243            for i in 0..cluster_count {
244                let count = pixel_count_sums[i];
245
246                if count == 0 {
247                    clusters[i] = Lab::new(0.0, 0.0, 0.0);
248
249                    continue;
250                }
251
252                let a = component_asums[i] / f64::from(count);
253                let b = component_bsums[i] / f64::from(count);
254                let c = component_csums[i] / f64::from(count);
255
256                clusters[i] = Lab::new(a, b, c);
257            }
258        }
259
260        let mut cluster_rgbs = vec![];
261        let mut cluster_populations = vec![];
262
263        for i in 0..cluster_count {
264            let count = pixel_count_sums[i];
265
266            if count == 0 {
267                continue;
268            }
269
270            let possible_new_cluster = PointProviderLab::lab_to_int(&clusters[i]);
271
272            if cluster_rgbs.contains(&possible_new_cluster) {
273                continue;
274            }
275
276            cluster_rgbs.push(possible_new_cluster);
277
278            cluster_populations.push(count);
279        }
280
281        #[cfg(feature = "std")]
282        Self::debug_log(format!(
283            "kmeans finished and generated {} clusters; {cluster_count} were requested",
284            cluster_rgbs.len()
285        ));
286
287        let mut input_pixel_to_cluster_pixel: IndexMap<Rgb, Rgb> = IndexMap::default();
288
289        #[cfg(feature = "std")]
290        let start_time = Instant::now();
291
292        for i in 0..pixels.len() {
293            let input_pixel = pixels[i];
294            let cluster_index = cluster_indices[i];
295            let cluster = clusters[cluster_index];
296            let cluster_pixel = PointProviderLab::lab_to_int(&cluster);
297
298            input_pixel_to_cluster_pixel.insert(input_pixel, cluster_pixel);
299        }
300
301        #[cfg(feature = "std")]
302        let time_elapsed = start_time.elapsed().as_millis();
303
304        #[cfg(feature = "std")]
305        Self::debug_log(format!("took {time_elapsed} ms to create input to cluster map"));
306
307        let mut color_to_count: IndexMap<Rgb, u32> = IndexMap::default();
308
309        for i in 0..cluster_rgbs.len() {
310            let key = cluster_rgbs[i];
311            let value = cluster_populations[i];
312
313            color_to_count.insert(key, value);
314        }
315
316        QuantizerResult {
317            color_to_count,
318            input_pixel_to_cluster_pixel,
319        }
320    }
321}

323fn fill_array<T>(count: usize, callback: impl Fn(usize) -> T) -> Vec<T> {
324    let mut results: Vec<T> = vec![];
325
326    for index in 0..count {
327        results.push(callback(index));
328    }
329
330    results
331}

8/// An intermediate concept between the key color for a UI theme, and a full
9/// color scheme. 5 tonal palettes are generated, all except one use the same
10/// hue as the key color, and all vary in chroma.
11#[derive(Debug, Hash, PartialEq, Eq)]
12#[deprecated = "Use `DynamicScheme` for color scheme generation. Use `CorePalettes` for core palettes container class"]
13pub struct CorePalette {
14    pub primary: TonalPalette,
15    pub secondary: TonalPalette,
16    pub tertiary: TonalPalette,
17    pub neutral: TonalPalette,
18    pub neutral_variant: TonalPalette,
19    pub error: TonalPalette,
20}
21
22impl CorePalette {
23    pub fn new(
24        primary: TonalPalette,
25        secondary: TonalPalette,
26        tertiary: TonalPalette,
27        neutral: TonalPalette,
28        neutral_variant: TonalPalette,
29        error: Option<TonalPalette>,
30    ) -> Self {
31        Self {
32            primary,
33            secondary,
34            tertiary,
35            neutral,
36            neutral_variant,
37            error: error.unwrap_or_else(|| TonalPalette::of(25.0, 84.0)),
38        }
39    }
40
41    /// Create a [`CorePalette`] from a source Rgb color.
42    pub fn of(rgb: Rgb) -> Self {
43        let cam = Cam16::from(rgb);
44        let (hue, chroma) = (cam.hue, cam.chroma);
45
46        Self::new(
47            TonalPalette::of(hue, 48.0_f64.max(chroma)),
48            TonalPalette::of(hue, 16.0),
49            TonalPalette::of(hue + 60.0, 24.0),
50            TonalPalette::of(hue, 4.0),
51            TonalPalette::of(hue, 8.0),
52            None,
53        )
54    }
55
56    /// Create a content [`CorePalette`] from a source Rgb color.
57    pub fn content_of(rgb: Rgb) -> Self {
58        let cam = Cam16::from(rgb);
59        let (hue, chroma) = (cam.hue, cam.chroma);
60
61        Self::new(
62            TonalPalette::of(hue, chroma),
63            TonalPalette::of(hue, chroma / 3.0),
64            TonalPalette::of(hue + 60.0, chroma / 2.0),
65            TonalPalette::of(hue, (chroma / 12.0).min(4.0)),
66            TonalPalette::of(hue, (chroma / 6.0).min(8.0)),
67            None,
68        )
69    }
70}
71
72impl fmt::Display for CorePalette {
73    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
74        write!(
75            f,
76            "primary{} secondary{} tertiary{} neutral{} neutral_variant{}",
77            self.primary, self.secondary, self.tertiary, self.neutral, self.neutral_variant
78        )
79    }

82/// Comprises foundational palettes to build a color scheme. Generated from a
83/// source color, these palettes will then be part of a [`DynamicScheme`]
84/// together with appearance preferences.
85///
86/// [`DynamicScheme`]: [crate::dynamic_color::dynamic_scheme::DynamicScheme]
87pub struct CorePalettes {
88    pub primary: TonalPalette,
89    pub secondary: TonalPalette,
90    pub tertiary: TonalPalette,
91    pub neutral: TonalPalette,
92    pub neutral_variant: TonalPalette,
93}

103    fn hash_value<T: Hash>(value: &T) -> u64 {
104        let mut hasher = AHasher::default();
105
106        value.hash(&mut hasher);
107
108        hasher.finish()
109    }

112    fn test_equals_and_hash() {
113        let core_palette_a = CorePalette::of(Rgb::from_u32(0x0000FF));
114        let core_palette_b = CorePalette::of(Rgb::from_u32(0x0000FF));
115        let core_palette_c = CorePalette::of(Rgb::from_u32(0x123456));
116
117        assert_eq!(core_palette_a, core_palette_b);
118        assert!(core_palette_b != core_palette_c);
119
120        assert_eq!(hash_value(&core_palette_a), hash_value(&core_palette_b));
121        assert!(hash_value(&core_palette_b) != hash_value(&core_palette_c));

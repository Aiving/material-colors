311impl From<Scheme> for Map<String, String> {
312    fn from(value: Scheme) -> Self {
313        let map: Map<String, Argb> = Map::from_iter(value);
314
315        map.into_iter()
316            .map(|(key, value)| (key, value.to_hex_with_pound()))
317            .collect()
318    }

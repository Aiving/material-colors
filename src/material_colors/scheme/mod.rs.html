292impl From<Scheme> for Map<String, String> {
293    fn from(value: Scheme) -> Self {
294        let map: Map<String, Rgb> = Map::from_iter(value);
295
296        map.into_iter().map(|(key, value)| (key, value.to_hex_with_pound())).collect()
297    }

15/// Design utilities using color temperature theory.
16///
17/// Analogous colors, complementary color, and cache to efficiently, lazily,
18/// generate data for calculations when needed.
19pub struct TemperatureCache {
20    input: Hct,
21
22    /// HCTs for all hues, with the same chroma/tone as the input.
23    /// Sorted from coldest first to warmest last.
24    hcts_by_temp: [Hct; 362],
25    /// HCTs for all hues, with the same chroma/tone as the input.
26    /// Sorted ascending, hue 0 to 360.
27    hcts_by_hue: [Hct; 362],
28    /// A Map with keys of HCTs in `hcts_by_temp`, values of raw temperature.
29    temps_by_hct: Map<Hct, f64>,
30    /// Relative temperature of the input color. See [`relative_temperature`].
31    ///
32    /// [`relative_temperature`]: Self::relative_temperature
33    input_relative_temperature: f64,
34    _complement: Option<Hct>,
35}
36
37impl TemperatureCache {
38    /// # Panics
39    ///
40    /// Will panic if there is no warmest HCT
41    pub const fn warmest(&self) -> &Hct {
42        &self.hcts_by_temp[361]
43    }
44
45    /// # Panics
46    ///
47    /// Will panic if there is no coldest HCT
48    pub const fn coldest(&self) -> &Hct {
49        &self.hcts_by_temp[0]
50    }
51
52    pub fn new(input: Hct) -> Self {
53        let chroma = input.get_chroma();
54        let tone = input.get_tone();
55
56        let hcts_by_hue = core::array::from_fn(|index| {
57            if index == 361 {
58                input
59            } else {
60                Hct::from(f64::from(index as i32), chroma, tone)
61            }
62        });
63
64        let temps_by_hct = hcts_by_hue.iter().map(|e| (*e, Self::raw_temperature(e))).collect();
65
66        let mut hcts_by_temp = hcts_by_hue;
67
68        hcts_by_temp.sort_by(|a, b| Self::sort_by_temp(&temps_by_hct, a, b));
69
70        let mut cache = Self {
71            input,
72            hcts_by_temp,
73            hcts_by_hue,
74            temps_by_hct,
75            input_relative_temperature: -1.0,
76            _complement: None,
77        };
78
79        cache.input_relative_temperature = {
80            let coldest = cache.coldest();
81            let warmest = cache.warmest();
82            let input = &cache.input;
83
84            let coldest_temp = cache.temps_by_hct[coldest];
85
86            let range = cache.temps_by_hct[warmest] - coldest_temp;
87            let difference_from_coldest = cache.temps_by_hct[input] - coldest_temp;
88
89            if range == 0.0 { 0.5 } else { difference_from_coldest / range }
90        };
91
92        cache
93    }

95    /// A set of colors with differing hues, equidistant in temperature.
96    ///
97    /// In art, this is usually described as a set of 5 colors on a color wheel
98    /// divided into 12 sections. This method allows provision of either of
99    /// those values.
100    ///
101    /// Behavior is undefined when `count` or `divisions` is 0.
102    /// When `divisions` < `count`, colors repeat.
103    ///
104    /// - `count`: The number of colors to return, includes the input color.
105    /// - `divisions`: The number of divisions on the color wheel.
106    pub fn analogous(&self, count: Option<i32>, divisions: Option<i32>) -> Vec<Hct> {
107        let count = count.unwrap_or(5);
108        let divisions = divisions.unwrap_or(12);
109        let start_hue = self.input.get_hue().round() as i32;
110
111        let start_hct = self.hcts_by_hue[start_hue as usize];
112        let mut last_temp = self.relative_temperature(&start_hct);
113        let mut all_colors = vec![start_hct];
114
115        let mut absolute_total_temp_delta = 0.0;
116
117        for i in 0..360 {
118            let hue = sanitize_degrees_double((start_hue + i).into());
119            let hct = self.hcts_by_hue[hue as usize];
120            let temp = self.relative_temperature(&hct);
121            let temp_delta = (temp - last_temp).abs();
122
123            last_temp = temp;
124            absolute_total_temp_delta += temp_delta;
125        }
126
127        let mut hue_addend = 1;
128        let temp_step = absolute_total_temp_delta / f64::from(divisions);
129
130        let mut total_temp_delta = 0.0;
131
132        last_temp = self.relative_temperature(&start_hct);
133
134        while all_colors.len() < divisions as usize {
135            let hue = sanitize_degrees_double((start_hue + hue_addend).into());
136            let hct = self.hcts_by_hue[hue as usize];
137            let temp = self.relative_temperature(&hct);
138            let temp_delta = (temp - last_temp).abs();
139
140            total_temp_delta += temp_delta;
141
142            let desired_total_temp_delta_for_index = all_colors.len() as f64 * temp_step;
143
144            let mut index_satisfied = total_temp_delta >= desired_total_temp_delta_for_index;
145            let mut index_addend = 1;
146
147            // Keep adding this hue to the answers until its temperature is
148            // insufficient. This ensures consistent behavior when there aren't
149            // [divisions] discrete steps between 0 and 360 in hue with [tempStep]
150            // delta in temperature between them.
151            //
152            // For example, white and black have no analogues: there are no other
153            // colors at T100/T0. Therefore, they should just be added to the array
154            // as answers.
155            while index_satisfied && all_colors.len() < divisions as usize {
156                all_colors.push(hct);
157
158                let desired_total_temp_delta_for_index = (all_colors.len() + index_addend) as f64 * temp_step;
159
160                index_satisfied = total_temp_delta >= desired_total_temp_delta_for_index;
161                index_addend += 1;
162            }
163
164            last_temp = temp;
165            hue_addend += 1;
166
167            if hue_addend > 360 {
168                while all_colors.len() < divisions as usize {
169                    all_colors.push(hct);
170                }
171
172                break;
173            }
174        }
175
176        let mut answers = vec![self.input];
177
178        // First, generate analogues from rotating counter-clockwise.
179        let increase_hue_count = ((f64::from(count) - 1.0) / 2.0).floor() as isize;
180
181        for i in 1..=increase_hue_count {
182            let mut index = 0_isize - i;
183
184            while index < 0 {
185                index += all_colors.len() as isize;
186            }
187
188            if index >= all_colors.len() as isize {
189                index %= all_colors.len() as isize;
190            }
191
192            answers.insert(0, all_colors[index as usize]);
193        }
194
195        // Second, generate analogues from rotating clockwise.
196        let decrease_hue_count = (count - (increase_hue_count as i32) - 1) as isize;
197
198        for i in 1..=decrease_hue_count {
199            let mut index = i;
200
201            while index < 0 {
202                index += all_colors.len() as isize;
203            }
204
205            if index >= all_colors.len() as isize {
206                index %= all_colors.len() as isize;
207            }
208
209            answers.push(all_colors[index as usize]);
210        }
211
212        answers
213    }

215    /// A color that complements the input color aesthetically.
216    ///
217    /// In art, this is usually described as being across the color wheel.
218    /// History of this shows intent as a color that is just as cool-warm as the
219    /// input color is warm-cool.
220    ///
221    /// # Panics
222    ///
223    /// Will panic if there is no coldest or warmest HCT
224    pub fn complement(&mut self) -> Hct {
225        if let Some(complement) = self._complement {
226            return complement;
227        }
228
229        let coldest_hct = self.coldest();
230        let warmest_hct = self.warmest();
231
232        let coldest_hue = coldest_hct.get_hue();
233        let coldest_temp = self.temps_by_hct[coldest_hct];
234
235        let warmest_hue = warmest_hct.get_hue();
236        let warmest_temp = self.temps_by_hct[warmest_hct];
237
238        let range = warmest_temp - coldest_temp;
239        let start_hue_is_coldest_to_warmest = Self::is_between(self.input.get_hue(), coldest_hue, warmest_hue);
240
241        let start_hue = if start_hue_is_coldest_to_warmest { warmest_hue } else { coldest_hue };
242
243        let end_hue = if start_hue_is_coldest_to_warmest { coldest_hue } else { warmest_hue };
244
245        let direction_of_rotation = 1.0_f64;
246        let mut smallest_error = 1000.0;
247        let hue = self.input.get_hue().round();
248        let mut answer = self.hcts_by_hue[hue as usize];
249
250        let complement_relative_temp = 1.0 - self.input_relative_temperature;
251
252        // Find the color in the other section, closest to the inverse percentile
253        // of the input color. This is the complement.
254        for hue_addend in 0..=360 {
255            let hue = sanitize_degrees_double(direction_of_rotation.mul_add(f64::from(hue_addend), start_hue));
256
257            if !Self::is_between(hue, start_hue, end_hue) {
258                continue;
259            }
260
261            let possible_answer = &self.hcts_by_hue[hue.round() as usize];
262            let relative_temp = (self.temps_by_hct[possible_answer] - coldest_temp) / range;
263            let error = (complement_relative_temp - relative_temp).abs();
264
265            if error < smallest_error {
266                smallest_error = error;
267                answer = *possible_answer;
268            }
269        }
270
271        self._complement = Some(answer);
272
273        answer
274    }

276    /// Temperature relative to all colors with the same chroma and tone.
277    /// Value on a scale from 0 to 1.
278    pub fn relative_temperature(&self, hct: &Hct) -> f64 {
279        let coldest = self.coldest();
280        let warmest = self.warmest();
281
282        let range = self.temps_by_hct[warmest] - self.temps_by_hct[coldest];
283        let difference_from_coldest = self.temps_by_hct[hct] - self.temps_by_hct[coldest];
284
285        // Handle when there's no difference in temperature between warmest and
286        // coldest: for example, at T100, only one color is available, white.
287        if range == 0.0 { 0.5 } else { difference_from_coldest / range }
288    }

290    fn sort_by_temp(temps_by_hct: &Map<Hct, f64>, this: &Hct, that: &Hct) -> Ordering {
291        let a = &temps_by_hct[this];
292        let b = &temps_by_hct[that];
293
294        a.partial_cmp(b).unwrap()
295    }

297    /// Determines if an angle is between two other angles, rotating clockwise.
298    pub fn is_between(angle: f64, a: f64, b: f64) -> bool {
299        if a < b { a <= angle && angle <= b } else { a <= angle || angle <= b }
300    }

302    /// Value representing cool-warm factor of a color.
303    /// Values below 0 are considered cool, above, warm.
304    ///
305    /// Color science has researched emotion and harmony, which art uses to
306    /// select colors. Warm-cool is the foundation of analogous and
307    /// complementary colors. See:
308    /// - Li-Chen Ou's Chapter 19 in Handbook of Color Psychology (2015).
309    /// - Josef Albers' Interaction of Color chapters 19 and 21.
310    ///
311    /// Implementation of Ou, Woodcock and Wright's algorithm, which uses
312    /// L*a*b*/LCH color space.
313    /// Return value has these properties:
314    /// - Values below 0 are cool, above 0 are warm.
315    /// - Lower bound: -0.52 - (chroma ^ 1.07 / 20). L*a*b* chroma is infinite.
316    ///   Assuming max of 130 chroma, -9.66.
317    /// - Upper bound: -0.52 + (chroma ^ 1.07 / 20). L*a*b* chroma is infinite.
318    ///   Assuming max of 130 chroma, 8.61.
319    pub fn raw_temperature(color: &Hct) -> f64 {
320        let lab = Lab::from(Rgb::from_ref(color));
321        let hue = sanitize_degrees_double(lab.b.atan2(lab.a).to_degrees());
322
323        let chroma = lab.a.hypot(lab.b);
324
325        (0.02 * chroma.powf(1.07)).mul_add((sanitize_degrees_double(hue - 50.0).to_radians()).cos(), -0.5)
326    }

336    #[test]
337    fn test_raw_temperature() {
338        let blue_hct = Hct::new(Rgb::from_u32(0x0000FF));
339        let red_hct = Hct::new(Rgb::from_u32(0xFF0000));
340        let green_hct = Hct::new(Rgb::from_u32(0x00FF00));
341        let white_hct = Hct::new(Rgb::from_u32(0xFFFFFF));
342        let black_hct = Hct::new(Rgb::from_u32(0x000000));
343
344        let blue_temp = TemperatureCache::raw_temperature(&blue_hct);
345        let red_temp = TemperatureCache::raw_temperature(&red_hct);
346        let green_temp = TemperatureCache::raw_temperature(&green_hct);
347        let white_temp = TemperatureCache::raw_temperature(&white_hct);
348        let black_temp = TemperatureCache::raw_temperature(&black_hct);
349
350        assert_approx_eq!(f64, -1.393, blue_temp, epsilon = 0.001);
351        assert_approx_eq!(f64, 2.351, red_temp, epsilon = 0.001);
352        assert_approx_eq!(f64, -0.267, green_temp, epsilon = 0.001);
353        assert_approx_eq!(f64, -0.5, white_temp, epsilon = 0.001);
354        assert_approx_eq!(f64, -0.5, black_temp, epsilon = 0.001);
355    }

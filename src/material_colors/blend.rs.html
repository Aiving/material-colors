10pub fn harmonize(design_color: Argb, source_color: Argb) -> Argb {
11    let from_hct: Hct = design_color.into();
12    let to_hct: Hct = source_color.into();
13
14    let difference_degrees = difference_degrees(from_hct.get_hue(), to_hct.get_hue());
15    let rotation_degrees = (difference_degrees * 0.5).min(15.0);
16
17    let output_hue = sanitize_degrees_double(rotation_degrees.mul_add(
18        rotate_direction(from_hct.get_hue(), to_hct.get_hue()),
19        from_hct.get_hue(),
20    ));
21
22    Hct::from(output_hue, from_hct.get_chroma(), from_hct.get_tone()).into()
23}

25pub fn hct_hue(from: Argb, to: Argb, amount: f64) -> Argb {
26    let ucs = cam16_ucs(from, to, amount);
27
28    let ucs_cam = Cam16::from(ucs);
29    let from_cam = Cam16::from(from);
30
31    let blended = Hct::from(ucs_cam.hue, from_cam.chroma, from.as_lstar());
32
33    blended.into()
34}

36pub fn cam16_ucs(from: Argb, to: Argb, amount: f64) -> Argb {
37    let from_cam = Cam16::from(from);
38    let to_cam = Cam16::from(to);
39
40    let from_j = from_cam.jstar;
41    let from_a = from_cam.astar;
42    let from_b = from_cam.bstar;
43
44    let to_j = to_cam.jstar;
45    let to_a = to_cam.astar;
46    let to_b = to_cam.bstar;
47
48    let (jstar, astar, bstar) = (
49        (to_j - from_j).mul_add(amount, from_j),
50        (to_a - from_a).mul_add(amount, from_a),
51        (to_b - from_b).mul_add(amount, from_b),
52    );
53
54    Cam16::from_ucs(jstar, astar, bstar).into()
55}

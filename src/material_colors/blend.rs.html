10pub fn harmonize(design_color: Rgb, source_color: Rgb) -> Rgb {
11    let from_hct: Hct = design_color.into();
12    let to_hct: Hct = source_color.into();
13
14    let difference_degrees = difference_degrees(from_hct.get_hue(), to_hct.get_hue());
15    let rotation_degrees = (difference_degrees * 0.5).min(15.0);
16
17    let output_hue = sanitize_degrees_double(rotation_degrees.mul_add(rotate_direction(from_hct.get_hue(), to_hct.get_hue()), from_hct.get_hue()));
18
19    Hct::from(output_hue, from_hct.get_chroma(), from_hct.get_tone()).into()
20}

22pub fn hct_hue(from: Rgb, to: Rgb, amount: f64) -> Rgb {
23    let ucs = cam16_ucs(from, to, amount);
24
25    let ucs_cam = Cam16::from(ucs);
26    let from_cam = Cam16::from(from);
27
28    let blended = Hct::from(ucs_cam.hue, from_cam.chroma, from.as_lstar());
29
30    blended.into()
31}

33pub fn cam16_ucs(from: Rgb, to: Rgb, amount: f64) -> Rgb {
34    let from_cam = Cam16::from(from);
35    let to_cam = Cam16::from(to);
36
37    let from_j = from_cam.jstar;
38    let from_a = from_cam.astar;
39    let from_b = from_cam.bstar;
40
41    let to_j = to_cam.jstar;
42    let to_a = to_cam.astar;
43    let to_b = to_cam.bstar;
44
45    let (jstar, astar, bstar) = (
46        (to_j - from_j).mul_add(amount, from_j),
47        (to_a - from_a).mul_add(amount, from_a),
48        (to_b - from_b).mul_add(amount, from_b),
49    );
50
51    Cam16::from_ucs(jstar, astar, bstar).into()
52}

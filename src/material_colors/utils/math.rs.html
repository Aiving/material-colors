5pub const fn signum(value: f64) -> f64 {
6    if value < 0.0 {
7        -1.0
8    } else if value == 0.0 {
9        0.0
10    } else {
11        1.0
12    }
13}

15pub const fn lerp(start: f64, stop: f64, amount: f64) -> f64 {
16    (1.0 - amount) * start + amount * stop
17}

19pub const fn sanitize_degrees_int(degrees: i32) -> u32 {
20    match degrees {
21        value if value < 0 => (value + 360) as u32,
22        value => value as u32 % 360,
23    }
24}
25
26pub const fn sanitize_degrees_double(degrees: f64) -> f64 {
27    match degrees {
28        value if value < 0.0 => value + 360.0,
29        value => value % 360.0,
30    }
31}

33pub const fn rotate_direction(from: f64, to: f64) -> f64 {
34    let increasing_difference = sanitize_degrees_double(to - from);
35
36    if increasing_difference <= 180.0 { 1.0 } else { -1.0 }
37}

39pub const fn difference_degrees(a: f64, b: f64) -> f64 {
40    180.0 - ((a - b).abs() - 180.0).abs()
41}

43pub fn matrix_multiply(row: [f64; 3], matrix: [[f64; 3]; 3]) -> [f64; 3] {
44    [
45        row[2].mul_add(matrix[0][2], row[0].mul_add(matrix[0][0], row[1] * matrix[0][1])),
46        row[2].mul_add(matrix[1][2], row[0].mul_add(matrix[1][0], row[1] * matrix[1][1])),
47        row[2].mul_add(matrix[2][2], row[0].mul_add(matrix[2][0], row[1] * matrix[2][1])),
48    ]
49}

161    // Original implementation for MathUtils.rotateDirection.
162    // Included here to test equivalence with new implementation.
163    fn old_rotate_direction(from: f64, to: f64) -> f64 {
164        let a = to - from;
165        let b = to - from + 360.0;
166        let c = to - from - 360.0;
167
168        let (a_abs, b_abs, c_abs) = (a.abs(), b.abs(), c.abs());
169
170        if a_abs <= b_abs && a_abs <= c_abs {
171            if a >= 0.0 { 1.0 } else { -1.0 }
172        } else if b_abs <= a_abs && b_abs <= c_abs {
173            if b >= 0.0 { 1.0 } else { -1.0 }
174        } else if c >= 0.0 {
175            1.0
176        } else {
177            -1.0
178        }
179    }

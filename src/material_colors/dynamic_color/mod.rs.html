32/// A color that adjusts itself based on UI state provided by `DynamicScheme`.
33///
34/// This color automatically adjusts to accommodate a desired contrast level, or
35/// other adjustments such as differing in light mode versus dark mode, or what
36/// the theme is, or what the color that produced the theme is, etc.
37///
38/// Colors without backgrounds do not change tone when contrast changes. Colors
39/// with backgrounds become closer to their background as contrast lowers, and
40/// further when contrast increases.
41///
42/// Prefer the static constructors. They provide a much more simple interface,
43/// such as requiring just a hexcode, or just a hexcode and a background.
44///
45/// Ultimately, each component necessary for calculating a color, adjusting it
46/// for a desired contrast level, and ensuring it has a certain lightness/tone
47/// difference from another color, is provided by a function that takes a
48/// `DynamicScheme` and returns a value. This ensures ultimate flexibility, any
49/// desired behavior of a color for any design system, but it usually
50/// unnecessary. See the default constructor for more information.
51pub struct DynamicColor {
52    pub name: String,
53    palette: Box<DynamicSchemeFnRef<TonalPalette>>,
54    tone: Box<fn(&DynamicScheme) -> f64>,
55    is_background: bool,
56    background: Option<Box<DynamicSchemeFn<DynamicColor>>>,
57    second_background: Option<Box<DynamicSchemeFn<DynamicColor>>>,
58    contrast_curve: Option<ContrastCurve>,
59    tone_delta_pair: Option<Box<DynamicSchemeFn<ToneDeltaPair>>>,
60}
61
62impl DynamicColor {
63    /// The base constructor for `DynamicColor`.
64    ///
65    /// _Strongly_ prefer using one of the convenience constructors. This class is
66    /// arguably too flexible to ensure it can support any scenario. Functional
67    /// arguments allow  overriding without risks that come with subclasses.
68    ///
69    /// For example, the default behavior of adjust tone at max contrast
70    /// to be at a 7.0 ratio with its background is principled and
71    /// matches accessibility guidance. That does not mean it's the desired
72    /// approach for _every_ design system, and every color pairing,
73    /// always, in every case.
74    ///
75    /// - Parameters:
76    ///   - `name`: The name of the dynamic color.
77    ///   - `palette`: Function that provides a [`TonalPalette`] given [`DynamicScheme`].
78    ///       A [`TonalPalette`] is defined by a hue and chroma, so this
79    ///       replaces the need to specify hue/chroma. By providing a tonal palette,
80    ///       when contrast adjustments are made, intended chroma can be preserved.
81    ///   - `tone`: Function that provides a tone, given a [`DynamicScheme`].
82    ///   - `isBackground`: Whether this dynamic color is a background, with
83    ///       some other color as the foreground.
84    ///   - `background`: The background of the dynamic color (as a function of a
85    ///       [`DynamicScheme`]), if it exists.
86    ///   - `secondBackground`: A second background of the dynamic color (as a function
87    ///       of a [`DynamicScheme`]), if it exists.
88    ///   - `contrastCurve`: A [`ContrastCurve`] object specifying how its contrast
89    ///       against its background should behave in various contrast levels options.
90    ///   - `toneDeltaPair`: A [`ToneDeltaPair`] object specifying a tone delta
91    ///       constraint between two colors. One of them must be the color being constructed.
92    ///
93    /// Unlikely to be useful unless a design system has some distortions
94    /// where colors that don't have a background/foreground relationship
95    /// don't want to have a formal relationship or a principled value for their
96    /// tone distance based on common contrast / tone delta values, yet, want
97    /// tone distance.
98    pub fn new<T: Into<String>>(
99        name: T,
100        palette: fn(&DynamicScheme) -> &TonalPalette,
101        tone: fn(&DynamicScheme) -> f64,
102        is_background: bool,
103        background: Option<fn(&DynamicScheme) -> Self>,
104        second_background: Option<fn(&DynamicScheme) -> Self>,
105        contrast_curve: Option<ContrastCurve>,
106        tone_delta_pair: Option<fn(&DynamicScheme) -> ToneDeltaPair>,
107    ) -> Self {
108        Self {
109            name: name.into(),
110            palette: Box::new(palette),
111            tone: Box::new(tone),
112            is_background,
113            background: background.map(Box::new),
114            second_background: second_background.map(Box::new),
115            contrast_curve,
116            tone_delta_pair: tone_delta_pair.map(Box::new),
117        }
118    }

146    /// - Parameter scheme: Defines the conditions of the user interface, for example,
147    ///   whether or not it is dark mode or light mode, and what the desired
148    ///   contrast level is.
149    /// - Returns: a tone, T in the HCT color space, that this `DynamicColor` is under
150    ///   the conditions in `scheme`.
151    pub fn get_tone(&self, scheme: &DynamicScheme) -> f64 {
152        let decreasing_contrast = scheme.contrast_level < 0.0;
153
154        // Case 1: dual foreground, pair of colors with delta constraint.
155        if let Some(tone_delta_pair) = &self.tone_delta_pair {
156            let pair = (tone_delta_pair)(scheme);
157            let role_a = pair.subject;
158            let role_b = pair.basis;
159            let delta = pair.delta;
160            let polarity = pair.polarity;
161            let stay_together = pair.stay_together;
162
163            let bg = self.background.as_ref().unwrap()(scheme);
164            let bg_tone = bg.get_tone(scheme);
165
166            let a_is_nearer = polarity == TonePolarity::Nearer
167                || (polarity == TonePolarity::Lighter && !scheme.is_dark)
168                || (polarity == TonePolarity::Darker && scheme.is_dark);
169            let nearer = if a_is_nearer { &role_a } else { &role_b };
170            let farther = if a_is_nearer { &role_b } else { &role_a };
171            let am_nearer = self.name == nearer.name;
172            let expansion_dir = if scheme.is_dark { 1.0 } else { -1.0 };
173
174            // 1st round: solve to min, each
175            let n_contrast = nearer
176                .contrast_curve
177                .as_ref()
178                .unwrap()
179                .get(scheme.contrast_level);
180            let f_contrast = farther
181                .contrast_curve
182                .as_ref()
183                .unwrap()
184                .get(scheme.contrast_level);
185
186            // If a color is good enough, it is not adjusted.
187            // Initial and adjusted tones for `nearer`
188            let n_initial_tone = (nearer.tone)(scheme);
189            let mut n_tone = if decreasing_contrast {
190                Self::foreground_tone(bg_tone, n_contrast)
191            } else if ratio_of_tones(bg_tone, n_initial_tone) >= n_contrast {
192                n_initial_tone
193            } else {
194                Self::foreground_tone(bg_tone, n_contrast)
195            };
196            // Initial and adjusted tones for `farther`
197            let f_initial_tone = (farther.tone)(scheme);
198            let mut f_tone = if decreasing_contrast {
199                Self::foreground_tone(bg_tone, f_contrast)
200            } else if ratio_of_tones(bg_tone, f_initial_tone) >= f_contrast {
201                f_initial_tone
202            } else {
203                Self::foreground_tone(bg_tone, f_contrast)
204            };
205
206            if (f_tone - n_tone) * expansion_dir >= delta {
207                // Good! Tones satisfy the constraint; no change needed.
208            } else {
209                // 2nd round: expand farther to match delta.
210                f_tone = delta.mul_add(expansion_dir, n_tone).clamp(0.0, 100.0);
211
212                if (f_tone - n_tone) * expansion_dir >= delta {
213                    // Good! Tones now satisfy the constraint; no change needed.
214                } else {
215                    // 3rd round: contract nearer to match delta.
216                    n_tone = delta.mul_add(-expansion_dir, f_tone).clamp(0.0, 100.0);
217                }
218            }
219
220            // Avoids the 50-59 awkward zone.
221            if (50.0..60.0).contains(&n_tone) {
222                // If `nearer` is in the awkward zone, move it away, together with
223                // `farther`.
224                if expansion_dir > 0.0 {
225                    n_tone = 60.0;
226
227                    f_tone = f_tone.max(delta.mul_add(expansion_dir, n_tone));
228                } else {
229                    n_tone = 49.0;
230
231                    f_tone = f_tone.min(delta.mul_add(expansion_dir, n_tone));
232                }
233            } else if (50.0..60.0).contains(&f_tone) {
234                if stay_together {
235                    // Fixes both, to avoid two colors on opposite sides of the "awkward
236                    // zone".
237                    if expansion_dir > 0.0 {
238                        n_tone = 60.0;
239
240                        f_tone = f_tone.max(delta.mul_add(expansion_dir, n_tone));
241                    } else {
242                        n_tone = 49.0;
243
244                        f_tone = f_tone.min(delta.mul_add(expansion_dir, n_tone));
245                    }
246                } else {
247                    // Not required to stay together; fixes just one.
248                    if expansion_dir > 0.0 {
249                        f_tone = 60.0;
250                    } else {
251                        f_tone = 49.0;
252                    }
253                }
254            }
255
256            // Returns `nTone` if this color is `nearer`, otherwise `fTone`.
257            if am_nearer {
258                n_tone
259            } else {
260                f_tone
261            }
262        } else {
263            // Case 2: No contrast pair; just solve for itself.
264            let mut answer = (self.tone)(scheme);
265
266            if let Some(background) = &self.background {
267                let bg_tone = background(scheme).get_tone(scheme);
268
269                let desired_ratio = self
270                    .contrast_curve
271                    .as_ref()
272                    .unwrap()
273                    .get(scheme.contrast_level);
274
275                if ratio_of_tones(bg_tone, answer) >= desired_ratio {
276                    // Don't "improve" what's good enough.
277                } else {
278                    // Rough improvement.
279                    answer = Self::foreground_tone(bg_tone, desired_ratio);
280                }
281
282                if decreasing_contrast {
283                    answer = Self::foreground_tone(bg_tone, desired_ratio);
284                }
285
286                if self.is_background && (50.0..60.0).contains(&answer) {
287                    // Must adjust
288                    if ratio_of_tones(49.0, bg_tone) >= desired_ratio {
289                        answer = 49.0;
290                    } else {
291                        answer = 60.0;
292                    }
293                }
294
295                if let Some(second_background) = &self.second_background {
296                    // Case 3: Adjust for dual backgrounds.
297
298                    let bg_tone1 = self.background.as_ref().unwrap()(scheme).get_tone(scheme);
299                    let bg_tone2 = second_background(scheme).get_tone(scheme);
300
301                    let upper = bg_tone1.max(bg_tone2);
302                    let lower = bg_tone1.min(bg_tone2);
303
304                    if ratio_of_tones(upper, answer) >= desired_ratio
305                        && ratio_of_tones(lower, answer) >= desired_ratio
306                    {
307                        return answer;
308                    }
309
310                    // The darkest light tone that satisfies the desired ratio,
311                    // or -1 if such ratio cannot be reached.
312                    let light_option = lighter(upper, desired_ratio);
313
314                    // The lightest dark tone that satisfies the desired ratio,
315                    // or -1 if such ratio cannot be reached.
316                    let dark_option = darker(lower, desired_ratio);
317
318                    // Tones suitable for the foreground.
319                    let mut availables: Vec<f64> = vec![];
320
321                    if (light_option - -1.0).abs() > f64::EPSILON {
322                        availables.push(light_option);
323                    }
324
325                    if (dark_option - -1.0).abs() > f64::EPSILON {
326                        availables.push(dark_option);
327                    }
328
329                    let prefers_light = Self::tone_prefers_light_foreground(bg_tone1)
330                        || Self::tone_prefers_light_foreground(bg_tone2);
331
332                    if prefers_light {
333                        return if light_option < 0.0 {
334                            100.0
335                        } else {
336                            light_option
337                        };
338                    }
339
340                    if availables.len() == 1 {
341                        return availables[0];
342                    }
343
344                    return if dark_option < 0.0 { 0.0 } else { dark_option };
345                }
346            }
347
348            answer
349        }
350    }

352    /// Given a background tone, find a foreground tone, while ensuring they reach
353    /// a contrast ratio that is as close to `ratio` as possible.
354    ///
355    /// - Parameters:
356    ///   - bgTone: Tone in HCT. Range is 0 to 100, undefined behavior when it falls
357    ///     outside that range.
358    ///   - ratio: The contrast ratio desired between `bgTone` and the return value.
359    ///
360    /// - Returns: The desired foreground tone.
361    pub fn foreground_tone(bg_tone: f64, ratio: f64) -> f64 {
362        let lighter_tone = lighter_unsafe(bg_tone, ratio);
363        let darker_tone = darker_unsafe(bg_tone, ratio);
364        let lighter_ratio = ratio_of_tones(lighter_tone, bg_tone);
365        let darker_ratio = ratio_of_tones(darker_tone, bg_tone);
366        let prefer_ligher = Self::tone_prefers_light_foreground(bg_tone);
367
368        if prefer_ligher {
369            // This handles an edge case where the initial contrast ratio is high
370            // (ex. 13.0), and the ratio passed to the function is that high ratio,
371            // and both the lighter and darker ratio fails to pass that ratio.
372            //
373            // This was observed with Tonal Spot's On Primary Container turning black
374            // momentarily between high and max contrast in light mode.
375            // PC's standard tone was T90, OPC's was T10, it was light mode, and the
376            // contrast value was 0.6568521221032331.
377            let negligible_difference = (lighter_ratio - darker_ratio).abs() < 0.1
378                && lighter_ratio < ratio
379                && darker_ratio < ratio;
380
381            if lighter_ratio >= ratio || lighter_ratio >= darker_ratio || negligible_difference {
382                lighter_tone
383            } else {
384                darker_tone
385            }
386        } else if darker_ratio >= ratio || darker_ratio >= lighter_ratio {
387            darker_tone
388        } else {
389            lighter_tone
390        }
391    }

393    /// Adjusts a tone such that white has 4.5 contrast, if the tone is
394    /// reasonably close to supporting it.
395    /// - Parameter tone: The tone to be adjusted.
396    /// - Returns: The (possibly updated) tone.
397    pub fn enable_light_foreground(tone: f64) -> f64 {
398        if Self::tone_prefers_light_foreground(tone) && !Self::tone_allows_light_foreground(tone) {
399            return 49.0;
400        }
401
402        tone
403    }

405    /// Returns whether `tone` prefers a light foreground.
406    ///
407    /// People prefer white foregrounds on ~T60-70. Observed over time, and also
408    /// by Andrew Somers during research for APCA.
409    ///
410    /// T60 used as to create the smallest discontinuity possible when skipping
411    /// down to T49 in order to ensure light foregrounds.
412    ///
413    /// Since `tertiaryContainer` in dark monochrome scheme requires a tone of
414    /// 60, it should not be adjusted. Therefore, 60 is excluded here.
415    ///
416    /// - Parameter tone: The tone to be judged.
417    /// - Returns: whether `tone` prefers a light foreground.
418    pub fn tone_prefers_light_foreground(tone: f64) -> bool {
419        tone.round() < 60.0
420    }
421
422    /// Returns whether `tone` can reach a contrast ratio of 4.5 with a lighter
423    /// color.
424    ///
425    /// - Parameter tone: The tone to be judged.
426    /// - Returns: whether `tone` allows a light foreground.
427    pub fn tone_allows_light_foreground(tone: f64) -> bool {
428        tone.round() <= 49.0
429    }

44
45impl CustomColorGroup {
46    /// Generate custom color group from source and target color
47    ///
48    /// @link <https://m3.material.io/styles/color/the-color-system/color-roles>
49    fn new(source: Argb, color: CustomColor) -> Self {
50        let mut value = color.value;
51
52        if color.blend {
53            value = harmonize(value, source);
54        }
55
56        #[allow(deprecated)]
57        let palette = CorePalette::of(value);
58        #[allow(deprecated)]
59        let tones = palette.primary;
60
61        Self {
62            color,
63            value,
64            light: ColorGroup {
65                color: tones.tone(40),
66                on_color: tones.tone(100),
67                color_container: tones.tone(90),
68                on_color_container: tones.tone(10),
69            },
70            dark: ColorGroup {
71                color: tones.tone(80),
72                on_color: tones.tone(20),
73                color_container: tones.tone(30),
74                on_color_container: tones.tone(90),
75            },
76        }
77    }
78}

202    #[must_use]
203    pub fn build(mut self) -> Theme {
204        #[allow(deprecated)]
205        let palette = CorePalette::of(self.source);
206
207        if self.color_match {
208            self.variant = Variant::Fidelity;
209        }
210
211        let mut light = DynamicScheme::by_variant(self.source, &self.variant, false, None);
212        let mut dark = DynamicScheme::by_variant(self.source, &self.variant, true, None);
213
214        if let Some(color) = self.primary {
215            let palette = TonalPalette::by_variant(&color.into(), &self.variant, &Palette::Primary);
216
217            light.primary_palette = palette;
218            dark.primary_palette = palette;
219        }
220
221        if let Some(color) = self.secondary {
222            let palette =
223                TonalPalette::by_variant(&color.into(), &self.variant, &Palette::Secondary);
224
225            light.secondary_palette = palette;
226            dark.secondary_palette = palette;
227        }
228
229        if let Some(color) = self.tertiary {
230            let palette =
231                TonalPalette::by_variant(&color.into(), &self.variant, &Palette::Tertiary);
232
233            light.tertiary_palette = palette;
234            dark.tertiary_palette = palette;
235        }
236
237        if let Some(color) = self.error {
238            let palette = TonalPalette::by_variant(&color.into(), &self.variant, &Palette::Error);
239
240            light.error_palette = palette;
241            dark.error_palette = palette;
242        }
243
244        if let Some(color) = self.neutral {
245            let palette = TonalPalette::by_variant(&color.into(), &self.variant, &Palette::Neutral);
246
247            light.neutral_palette = palette;
248            dark.neutral_palette = palette;
249        }
250
251        if let Some(color) = self.neutral_variant {
252            let palette =
253                TonalPalette::by_variant(&color.into(), &self.variant, &Palette::NeutralVariant);
254
255            light.neutral_variant_palette = palette;
256            dark.neutral_variant_palette = palette;
257        }
258
259        Theme {
260            source: self.source,
261            schemes: Schemes {
262                light: light.into(),
263                dark: dark.into(),
264            },
265            #[allow(deprecated)]
266            palettes: Palettes {
267                primary: palette.primary,
268                secondary: palette.secondary,
269                tertiary: palette.tertiary,
270                neutral: palette.neutral,
271                neutral_variant: palette.neutral_variant,
272                error: palette.error,
273            },
274            custom_colors: self
275                .custom_colors
276                .into_iter()
277                .map(|color| CustomColorGroup::new(self.source, color))
278                .collect(),
279        }
280    }

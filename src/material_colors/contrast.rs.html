6/// Returns a contrast ratio, which ranges from 1 to 21.
7///
8/// - `toneA`: Tone between 0 and 100. Values outside will be clamped.
9/// - `toneB`: Tone between 0 and 100. Values outside will be clamped.
10pub fn ratio_of_tones(tone_a: f64, tone_b: f64) -> f64 {
11    let tone_a = tone_a.clamp(0.0, 100.0);
12    let tone_b = tone_b.clamp(0.0, 100.0);
13
14    ratio_of_ys(y_from_lstar(tone_a), y_from_lstar(tone_b))
15}
16
17fn ratio_of_ys(y1: f64, y2: f64) -> f64 {
18    let lighter = if y1 > y2 { y1 } else { y2 };
19    let darker = if (lighter - y2).abs() < f64::EPSILON { y1 } else { y2 };
20
21    (lighter + 5.0) / (darker + 5.0)
22}

24/// Returns a tone >= `tone` that ensures `ratio`. Return value is between 0 and
25/// 100. Returns -1 if `ratio` cannot be achieved with `tone`.
26///
27/// - `tone`: Tone return value must contrast with. Range is 0 to 100. Invalid
28///   values will result in -1 being returned.
29/// - `ratio`: Contrast ratio of return value and `tone`. Range is 1 to 21,
30///   invalid values have undefined behavior.
31pub fn lighter(tone: f64, ratio: f64) -> f64 {
32    if !(0.0..=100.0).contains(&tone) {
33        return -1.0;
34    }
35
36    let dark_y = y_from_lstar(tone);
37    let light_y = ratio.mul_add(dark_y + 5.0, -5.0);
38    let real_contrast = ratio_of_ys(light_y, dark_y);
39    let delta = (real_contrast - ratio).abs();
40
41    if real_contrast < ratio && delta > 0.04 {
42        return -1.0;
43    }
44
45    // Ensure gamut mapping, which requires a 'range' on tone, will still result
46    // the correct ratio by darkening slightly.
47    let return_value = lstar_from_y(light_y) + 0.4;
48
49    if !(0.0..=100.0).contains(&return_value) {
50        return -1.0;
51    }
52
53    return_value
54}

56/// Returns a tone <= `tone` that ensures `ratio`. Return value is between 0 and
57/// 100. Returns -1 if `ratio` cannot be achieved with `tone`.
58///
59/// - `tone`: Tone return value must contrast with. Range is 0 to 100. Invalid
60///   values will result in -1 being returned.
61/// - `ratio`: Contrast ratio of return value and `tone`. Range is 1 to 21,
62///   invalid values have undefined behavior.
63pub fn darker(tone: f64, ratio: f64) -> f64 {
64    if !(0.0..=100.0).contains(&tone) {
65        return -1.0;
66    }
67
68    let light_y = y_from_lstar(tone);
69    let dark_y = ((light_y + 5.0) / ratio) - 5.0;
70    let real_contrast = ratio_of_ys(light_y, dark_y);
71
72    let delta = (real_contrast - ratio).abs();
73
74    if real_contrast < ratio && delta > 0.04 {
75        return -1.0;
76    }
77
78    // Ensure gamut mapping, which requires a 'range' on tone, will still result
79    // the correct ratio by darkening slightly.
80    let return_value = lstar_from_y(dark_y) - 0.4;
81
82    if !(0.0..=100.0).contains(&return_value) {
83        return -1.0;
84    }
85
86    return_value
87}

89/// Returns a tone >= `tone` that ensures `ratio`. Return value is between 0 and
90/// 100. Returns 100 if `ratio` cannot be achieved with `tone`.
91///
92/// This method is unsafe because the returned value is guaranteed to be in
93/// bounds for tone, i.e. between 0 and 100. However, that value may not reach
94/// the `ratio` with `tone`. For example, there is no color lighter than T100.
95///
96/// - `tone`: Tone return value must contrast with. Range is 0 to 100. Invalid
97///   values will result in 100 being returned.
98/// - `ratio`: Desired contrast ratio of return value and tone parameter. Range
99///   is 1 to 21, invalid values have undefined behavior.
100pub fn lighter_unsafe(tone: f64, ratio: f64) -> f64 {
101    let lighter_safe = lighter(tone, ratio);
102
103    if lighter_safe < 0.0 { 100.0 } else { lighter_safe }
104}

106/// Returns a tone <= `tone` that ensures `ratio`. Return value is between 0 and
107/// 100. Returns 0 if `ratio` cannot be achieved with `tone`.
108///
109/// This method is unsafe because the returned value is guaranteed to be in
110/// bounds for tone, i.e. between 0 and 100. However, that value may not reach
111/// the `ratio` with `tone`. For example, there is no color darker than T0.
112///
113/// - `tone`: Tone return value must contrast with. Range is 0 to 100. Invalid
114///   values will result in 0 being returned.
115/// - `ratio`: Desired contrast ratio of return value and tone parameter. Range
116///   is 1 to 21, invalid values have undefined behavior.
117pub fn darker_unsafe(tone: f64, ratio: f64) -> f64 {
118    let darker_safe = darker(tone, ratio);
119
120    if darker_safe < 0.0 { 0.0 } else { darker_safe }
121}

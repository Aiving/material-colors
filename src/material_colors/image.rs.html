44impl ImageReader {
45    pub fn read<T>(data: T) -> Result<Image>
46    where
47        T: AsRef<[u8]>,
48    {
49        let data = Reader::new(Cursor::new(data))
50            .with_guessed_format()?
51            .decode()
52            .expect("failed to decode image")
53            .into_rgb8();
54
55        Ok(Image::new(data))
56    }
57
58    pub fn open<P>(path: P) -> Result<Image>
59    where
60        P: AsRef<Path>,
61    {
62        let data = Reader::open(path)?.with_guessed_format()?.decode().expect("failed to decode image").into_rgb8();
63
64        Ok(Image::new(data))
65    }

67    /// Get the source color from an image.
68    ///
69    /// `image` A struct that implements the [`AsPixels`] trait
70    ///
71    /// Returns source color - the color most suitable for creating a UI theme
72    pub fn extract_color<I>(image: &I) -> Rgb
73    where
74        I: AsPixels,
75    {
76        let pixels = image.as_pixels();
77        let result = QuantizerCelebi::quantize(&pixels, 128);
78        let ranked = Score::score(&result.color_to_count, None, None, None);
79
80        ranked[0]
81    }
